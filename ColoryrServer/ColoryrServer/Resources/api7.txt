using MySql.Data.MySqlClient;
using System;
using System.ComponentModel;
using System.ComponentModel;
using System.Data;
using System.Data.Common;

namespace MySql.Data.MySqlClient
{
    //
    // 摘要:
    //     Represents a SQL statement to execute against a MySQL database. This class cannot
    //     be inherited.
    //
    // 言论：
    //     MySqlCommand features the following methods for executing commands at a MySQL
    //     database:
    //     Item C Description C
    //     ExecuteReader C Executes commands that return rows.
    //     ExecuteNonQuery C Executes commands such as SQL INSERT, DELETE, and UPDATE statements.
    //     ExecuteScalar C Retrieves a single value (for example, an aggregate value) from
    //     a database.
    //     You can reset the CommandText property and reuse the MySqlCommand object. However,
    //     you must close the MySqlDataReader before you can execute a new or previous command.
    //     If a MySqlException is generated by the method executing a MySqlCommand, the
    //     MySqlConnection remains open. It is the responsibility of the programmer to close
    //     the connection. Using the '@' symbol for paramters is now the preferred approach
    //     although the old pattern of using '?' is still supported. Please be aware though
    //     that using '@' can cause conflicts when user variables are also used. To help
    //     with this situation please see the documentation on the 'allow user variables'
    //     connection string option. The 'old syntax' connection string option has now been
    //     deprecated.
    [DesignerCategory("Code")]
    public sealed class MySqlCommand : DbCommand, IDisposable, ICloneable
    {
        //
        // 摘要:
        //     Initializes a new instance of the MySqlCommand class.
        //
        // 言论：
        //     The base constructor initializes all fields to their default values. The following
        //     table shows initial property values for an instance of MySql.Data.MySqlClient.MySqlCommand.
        //     Properties C Initial Value C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandText C empty string ("") C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandTimeout C 0 C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandType C CommandType.Text C
        //     MySql.Data.MySqlClient.MySqlCommand.Connection C Null C
        //     You can change the value for any of these properties through a separate call
        //     to the property.
        public MySqlCommand();
        //
        // 摘要:
        //     Initializes a new instance of the MySql.Data.MySqlClient.MySqlCommand class with
        //     the text of the query.
        //
        // 参数:
        //   cmdText:
        //     The text of the query.
        //
        // 言论：
        //     When an instance of MySql.Data.MySqlClient.MySqlCommand is created, the following
        //     read/write properties are set to initial values.
        //     Properties C Initial Value C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandText C cmdText C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandTimeout C 0 C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandType C CommandType.Text C
        //     MySql.Data.MySqlClient.MySqlCommand.Connection C Null C
        //     You can change the value for any of these properties through a separate call
        //     to the property.
        public MySqlCommand(string cmdText);
        //
        // 摘要:
        //     Initializes a new instance of the MySql.Data.MySqlClient.MySqlCommand class with
        //     the text of the query and a MySql.Data.MySqlClient.MySqlConnection.
        //
        // 参数:
        //   cmdText:
        //     The text of the query.
        //
        //   connection:
        //     A MySql.Data.MySqlClient.MySqlConnection that represents the connection to an
        //     instance of SQL Server.
        //
        // 言论：
        //     When an instance of MySql.Data.MySqlClient.MySqlCommand is created, the following
        //     read/write properties are set to initial values.
        //     Properties C Initial Value C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandText C cmdText C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandTimeout C 0 C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandType C CommandType.Text C
        //     MySql.Data.MySqlClient.MySqlCommand.Connection C connection C
        //     You can change the value for any of these properties through a separate call
        //     to the property.
        public MySqlCommand(string cmdText, MySqlConnection connection);
        //
        // 摘要:
        //     Initializes a new instance of the MySql.Data.MySqlClient.MySqlCommand class with
        //     the text of the query, a MySql.Data.MySqlClient.MySqlConnection, and the MySql.Data.MySqlClient.MySqlTransaction.
        //
        // 参数:
        //   cmdText:
        //     The text of the query.
        //
        //   connection:
        //     A MySql.Data.MySqlClient.MySqlConnection that represents the connection to an
        //     instance of SQL Server.
        //
        //   transaction:
        //     The MySql.Data.MySqlClient.MySqlTransaction in which the MySql.Data.MySqlClient.MySqlCommand
        //     executes.
        //
        // 言论：
        //     When an instance of MySql.Data.MySqlClient.MySqlCommand is created, the following
        //     read/write properties are set to initial values.
        //     Properties C Initial Value C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandText C cmdText C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandTimeout C 0 C
        //     MySql.Data.MySqlClient.MySqlCommand.CommandType C CommandType.Text C
        //     MySql.Data.MySqlClient.MySqlCommand.Connection C connection C
        //     You can change the value for any of these properties through a separate call
        //     to the property.
        public MySqlCommand(string cmdText, MySqlConnection connection, MySqlTransaction transaction);

        ~MySqlCommand();

        //
        // 摘要:
        //     Gets or sets the wait time before terminating the attempt to execute a command
        //     and generating an error.
        //
        // 值:
        //     The time (in seconds) to wait for the command to execute. The default is 30 seconds.
        //
        // 言论：
        //     CommandTimeout is dependent on the ability of MySQL to cancel an executing query.
        //     Because of this, CommandTimeout is only supported when connected to MySQL version
        //     5.0.0 or higher.
        [Category("Misc")]
        [DefaultValue(30)]
        [Description("Time to wait for command to execute")]
        public override int CommandTimeout { get; set; }
        //
        // 摘要:
        //     Gets or sets a value indicating how the MySql.Data.MySqlClient.MySqlCommand.CommandText
        //     property is to be interpreted.
        //
        // 值:
        //     One of the System.Data.CommandType values. The default is Text.
        //
        // 言论：
        //     When you set the CommandType property to StoredProcedure, you should set the
        //     MySql.Data.MySqlClient.MySqlCommand.CommandText property to the name of the stored
        //     procedure. The command executes this stored procedure when you call one of the
        //     Execute methods.
        [Category("Data")]
        public override CommandType CommandType { get; set; }
        //
        // 摘要:
        //     Gets a boolean value that indicates whether the Prepared method has been called.
        [Browsable(false)]
        public bool IsPrepared { get; }
        //
        // 摘要:
        //     Gets or sets the MySql.Data.MySqlClient.MySqlConnection used by this instance
        //     of the MySql.Data.MySqlClient.MySqlCommand.
        //
        // 值:
        //     The connection to a data source. The default value is a null reference (Nothing
        //     in Visual Basic).
        //
        // 言论：
        //     If you set Connection while a transaction is in progress and the MySql.Data.MySqlClient.MySqlCommand.Transaction
        //     property is not null, an System.InvalidOperationException is generated. If the
        //     Transaction property is not null and the transaction has already been committed
        //     or rolled back, Transaction is set to null.
        [Category("Behavior")]
        [Description("Connection used by the command")]
        public MySqlConnection Connection { get; set; }
        //
        // 摘要:
        //     Get the MySql.Data.MySqlClient.MySqlParameterCollection
        //
        // 值:
        //     The parameters of the SQL statement or stored procedure. The default is an empty
        //     collection.
        //
        // 言论：
        //     Connector/NET does not support unnamed parameters. Every parameter added to the
        //     collection must have an associated name.
        [Category("Data")]
        [Description("The parameters collection")]
        [DesignerSerializationVisibility(DesignerSerializationVisibility.Content)]
        public MySqlParameterCollection Parameters { get; }
        //
        // 摘要:
        //     Gets or sets the seconds for how long a TableDirect result should be cached.
        public int CacheAge { get; set; }
        //
        // 摘要:
        //     Gets or sets a boolean value that indicates whether caching is enabled.
        public bool EnableCaching { get; set; }
        //
        // 摘要:
        //     Gets or sets the SQL statement to execute at the data source.
        //
        // 值:
        //     The SQL statement or stored procedure to execute. The default is an empty string.
        //
        // 言论：
        //     When the MySql.Data.MySqlClient.MySqlCommand.CommandType property is set to StoredProcedure,
        //     the CommandText property should be set to the name of the stored procedure. The
        //     user may be required to use escape character syntax if the stored procedure name
        //     contains any special characters. The command executes this stored procedure when
        //     you call one of the Execute methods. Starting with Connector/NET 5.0, having
        //     both a stored function and stored procedure with the same name in the same database
        //     is not supported. It is suggested that you provide unqiue names for your stored
        //     routines.
        [Category("Data")]
        [Description("Command text to execute")]
        public override string CommandText { get; set; }
        //
        // 摘要:
        //     Gets or sets how command results are applied to the DataRow when used by the
        //     Update method of the DbDataAdapter.
        public override UpdateRowSource UpdatedRowSource { get; set; }
        //
        // 摘要:
        //     Gets or sets a value indicating whether the command object should be visible
        //     in a Windows Form Designer control.
        [Browsable(false)]
        public override bool DesignTimeVisible { get; set; }
        //
        // 摘要:
        //     Gets or sets the MySql.Data.MySqlClient.MySqlTransaction within which the MySql.Data.MySqlClient.MySqlCommand
        //     executes.
        //
        // 值:
        //     The MySql.Data.MySqlClient.MySqlTransaction. The default value is a null reference
        //     (Nothing in Visual Basic).
        //
        // 言论：
        //     You cannot set the Transaction property if it is already set to a specific value,
        //     and the command is in the process of executing. If you set the transaction property
        //     to a MySql.Data.MySqlClient.MySqlTransaction object that is not connected to
        //     the same MySql.Data.MySqlClient.MySqlConnection as the MySql.Data.MySqlClient.MySqlCommand
        //     object, an exception will be thrown the next time you attempt to execute a statement.
        [Browsable(false)]
        public MySqlTransaction Transaction { get; set; }
        //
        // 摘要:
        //     Gets the last inserted id.
        [Browsable(false)]
        public long LastInsertedId { get; }
        protected override DbConnection DbConnection { get; set; }
        protected override DbTransaction DbTransaction { get; set; }
        protected override DbParameterCollection DbParameterCollection { get; }

        //
        // 摘要:
        //     Initiates the asynchronous execution of the SQL statement or stored procedure
        //     that is described by this MySql.Data.MySqlClient.MySqlCommand.
        //
        // 返回结果:
        //     An System.IAsyncResult that can be used to poll or wait for results, or both;
        //     this value is also needed when invoking MySql.Data.MySqlClient.MySqlCommand.EndExecuteNonQuery(System.IAsyncResult),
        //     which returns the number of affected rows.
        public IAsyncResult BeginExecuteNonQuery();
        //
        // 摘要:
        //     Initiates the asynchronous execution of the SQL statement or stored procedure
        //     that is described by this MySql.Data.MySqlClient.MySqlCommand.
        //
        // 参数:
        //   callback:
        //     An System.AsyncCallback delegate that is invoked when the command's execution
        //     has completed. Pass a null reference (Nothing in Visual Basic) to indicate that
        //     no callback is required.
        //
        //   stateObject:
        //     A user-defined state object that is passed to the callback procedure. Retrieve
        //     this object from within the callback procedure using the System.IAsyncResult.AsyncState
        //     property.
        //
        // 返回结果:
        //     An System.IAsyncResult that can be used to poll or wait for results, or both;
        //     this value is also needed when invoking MySql.Data.MySqlClient.MySqlCommand.EndExecuteNonQuery(System.IAsyncResult),
        //     which returns the number of affected rows.
        public IAsyncResult BeginExecuteNonQuery(AsyncCallback callback, object stateObject);
        //
        // 摘要:
        //     Initiates the asynchronous execution of the SQL statement or stored procedure
        //     that is described by this MySql.Data.MySqlClient.MySqlCommand using one of the
        //     CommandBehavior values.
        //
        // 参数:
        //   behavior:
        //     One of the System.Data.CommandBehavior values, indicating options for statement
        //     execution and data retrieval.
        //
        // 返回结果:
        //     An System.IAsyncResult that can be used to poll, wait for results, or both; this
        //     value is also needed when invoking EndExecuteReader, which returns a MySql.Data.MySqlClient.MySqlDataReader
        //     instance that can be used to retrieve the returned rows.
        public IAsyncResult BeginExecuteReader(CommandBehavior behavior);
        //
        // 摘要:
        //     Initiates the asynchronous execution of the SQL statement or stored procedure
        //     that is described by this MySql.Data.MySqlClient.MySqlCommand, and retrieves
        //     one or more result sets from the server.
        //
        // 返回结果:
        //     An System.IAsyncResult that can be used to poll, wait for results, or both; this
        //     value is also needed when invoking EndExecuteReader, which returns a MySql.Data.MySqlClient.MySqlDataReader
        //     instance that can be used to retrieve the returned rows.
        public IAsyncResult BeginExecuteReader();
        //
        // 摘要:
        //     Attempts to cancel the execution of a currently active command
        //
        // 言论：
        //     Cancelling a currently active query only works with MySQL versions 5.0.0 and
        //     higher.
        public override void Cancel();
        //
        // 摘要:
        //     Creates a clone of this MySql.Data.MySqlClient.MySqlCommand object. CommandText,
        //     Connection, and Transaction properties are included as well as the entire parameter
        //     list.
        //
        // 返回结果:
        //     The cloned MySql.Data.MySqlClient.MySqlCommand object.
        public object Clone();
        //
        // 摘要:
        //     Creates a new instance of a MySql.Data.MySqlClient.MySqlParameter object.
        //
        // 返回结果:
        //     A MySql.Data.MySqlClient.MySqlParameter object.
        //
        // 言论：
        //     This method is a strongly-typed version of System.Data.IDbCommand.CreateParameter.
        public MySqlParameter CreateParameter();
        public void Dispose();
        //
        // 摘要:
        //     Finishes asynchronous execution of a SQL statement.
        //
        // 参数:
        //   asyncResult:
        //     The System.IAsyncResult returned by the call to MySql.Data.MySqlClient.MySqlCommand.BeginExecuteNonQuery.
        public int EndExecuteNonQuery(IAsyncResult asyncResult);
        //
        // 摘要:
        //     Finishes asynchronous execution of a SQL statement, returning the requested MySql.Data.MySqlClient.MySqlDataReader.
        //
        // 参数:
        //   result:
        //     The System.IAsyncResult returned by the call to MySql.Data.MySqlClient.MySqlCommand.BeginExecuteReader.
        //
        // 返回结果:
        //     A MySqlDataReader object that can be used to retrieve the requested rows.
        public MySqlDataReader EndExecuteReader(IAsyncResult result);
        //
        // 摘要:
        //     Executes a SQL statement against the connection and returns the number of rows
        //     affected.
        //
        // 返回结果:
        //     Number of rows affected
        //
        // 言论：
        //     You can use ExecuteNonQuery to perform any type of database operation, however
        //     any resultsets returned will not be available. Any output parameters used in
        //     calling a stored procedure will be populated with data and can be retrieved after
        //     execution is complete. For UPDATE, INSERT, and DELETE statements, the return
        //     value is the number of rows affected by the command. For all other types of statements,
        //     the return value is -1.
        public override int ExecuteNonQuery();
        //
        // 摘要:
        //     Sends the MySql.Data.MySqlClient.MySqlCommand.CommandText to the Connection,
        //     and builds a MySql.Data.MySqlClient.MySqlDataReader using one of the System.Data.CommandBehavior
        //     values.
        //
        // 参数:
        //   behavior:
        //     One of the System.Data.CommandBehavior values.
        //
        // 返回结果:
        //     A MySql.Data.MySqlClient.MySqlDataReader object.
        //
        // 言论：
        //     When the MySql.Data.MySqlClient.MySqlCommand.CommandType property is set to StoredProcedure,
        //     the MySql.Data.MySqlClient.MySqlCommand.CommandText property should be set to
        //     the name of the stored procedure. The command executes this stored procedure
        //     when you call ExecuteReader.
        //     The MySql.Data.MySqlClient.MySqlDataReader supports a special mode that enables
        //     large binary values to be read efficiently. For more information, see the SequentialAccess
        //     setting for System.Data.CommandBehavior.
        //     While the MySql.Data.MySqlClient.MySqlDataReader is in use, the associated MySql.Data.MySqlClient.MySqlConnection
        //     is busy serving the MySqlDataReader. While in this state, no other operations
        //     can be performed on the MySqlConnection other than closing it. This is the case
        //     until the MySql.Data.MySqlClient.MySqlDataReader.Close method of the MySqlDataReader
        //     is called. If the MySqlDataReader is created with CommandBehavior set to CloseConnection,
        //     closing the MySqlDataReader closes the connection automatically.
        //     When calling ExecuteReader with the SingleRow behavior, you should be aware that
        //     using a limit clause in your SQL will cause all rows (up to the limit given)
        //     to be retrieved by the client. The MySql.Data.MySqlClient.MySqlDataReader.Read
        //     method will still return false after the first row but pulling all rows of data
        //     into the client will have a performance impact. If the limit clause is not necessary,
        //     it should be avoided.
        public MySqlDataReader ExecuteReader(CommandBehavior behavior);
        //
        // 摘要:
        //     Sends the MySql.Data.MySqlClient.MySqlCommand.CommandText to the Connection and
        //     builds a MySql.Data.MySqlClient.MySqlDataReader.
        //
        // 返回结果:
        //     A MySql.Data.MySqlClient.MySqlDataReader object.
        //
        // 言论：
        //     When the MySql.Data.MySqlClient.MySqlCommand.CommandType property is set to StoredProcedure,
        //     the MySql.Data.MySqlClient.MySqlCommand.CommandText property should be set to
        //     the name of the stored procedure. The command executes this stored procedure
        //     when you call ExecuteReader.
        //     While the MySql.Data.MySqlClient.MySqlDataReader is in use, the associated MySql.Data.MySqlClient.MySqlConnection
        //     is busy serving the MySqlDataReader. While in this state, no other operations
        //     can be performed on the MySqlConnection other than closing it. This is the case
        //     until the MySql.Data.MySqlClient.MySqlDataReader.Close method of the MySqlDataReader
        //     is called.
        public MySqlDataReader ExecuteReader();
        //
        // 摘要:
        //     Executes the query, and returns the first column of the first row in the result
        //     set returned by the query. Extra columns or rows are ignored.
        //
        // 返回结果:
        //     The first column of the first row in the result set, or a null reference if the
        //     result set is empty
        //
        // 言论：
        //     Use the ExecuteScalar method to retrieve a single value (for example, an aggregate
        //     value) from a database. This requires less code than using the MySql.Data.MySqlClient.MySqlCommand.ExecuteReader
        //     method, and then performing the operations necessary to generate the single value
        //     using the data returned by a MySql.Data.MySqlClient.MySqlDataReader
        public override object ExecuteScalar();
        //
        // 摘要:
        //     Creates a prepared version of the command on an instance of MySQL Server.
        //
        // 言论：
        //     Prepared statements are only supported on MySQL version 4.1 and higher. Calling
        //     prepare while connected to earlier versions of MySQL will succeed but will execute
        //     the statement in the same way as unprepared.
        public override void Prepare();
        protected override DbParameter CreateDbParameter();
        protected override void Dispose(bool disposing);
        protected override DbDataReader ExecuteDbDataReader(CommandBehavior behavior);
    }
